### Code
**Design Code**
```verilog
module decade_counter (
    input wire clk,       // Clock input
    input wire rst_n,     // Active-low synchronous reset
    output reg [3:0] count // 4-bit output
);

    always @(posedge clk) begin
        if (!rst_n) begin
            count <= 4'b0000; // Reset counter to 0
        end else if (count == 4'd9) begin
            count <= 4'b0000; // Wrap around to 0 after reaching 9
        end else begin
            count <= count + 1'b1; // Increment count
        end
    end

endmodule
```

### Code
**Design Code**
```verilog
module counter_4bit (
    input wire clk,
    input wire rst_n,
    input wire en,          // Enable counting
    output reg [3:0] count,
    output wire tc          // Terminal Count (High when count is 15 and enabled)
);

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            count <= 4'b0000;
        end else if (en) begin
            count <= count + 1'b1;
        end
    end

    // TC is high only when counter reaches max value (15) AND it is enabled
    assign tc = (count == 4'b1111) && en;

endmodule

module cascaded_8bit_counter (
    input wire clk,
    input wire rst_n,
    input wire en,
    output wire [7:0] count
);

    wire tc_low; // Carries the terminal count from lower 4-bits to upper 4-bits

    // Lower 4 bits (LSN) - Always enabled if top-level enable is high
    counter_4bit lower_nibble (
        .clk(clk),
        .rst_n(rst_n),
        .en(en),
        .count(count[3:0]),
        .tc(tc_low)
    );

    // Upper 4 bits (MSN) - Only increments when lower nibble hits 15 (tc_low is high)
    counter_4bit upper_nibble (
        .clk(clk),
        .rst_n(rst_n),
        .en(tc_low),
        .count(count[7:4]),
        .tc() // Leave unconnected if not cascading further
    );

endmodule
```

**Design Code**
```verilog
`timescale 1ns / 1ps
module counter_4bit (
    input wire clk,
    input wire rst_n,
    input wire en,          
    output reg [3:0] count,
    output wire tc          
);
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            count <= 4'b0000;
        end else if (en) begin
            count <= count + 1'b1;
        end
    end

    assign tc = (count == 4'b1111) && en;
endmodule

module seven_seg_decoder (
    input wire [3:0] bin_in,     
    output reg [6:0] seg_out     // Mapping: [g, f, e, d, c, b, a]
);
    always @(*) begin
        case (bin_in)
            4'h0: seg_out = 7'b0111111; 
            4'h1: seg_out = 7'b0000110; 
            4'h2: seg_out = 7'b1011011; 
            4'h3: seg_out = 7'b1001111; 
            4'h4: seg_out = 7'b1100110; 
            4'h5: seg_out = 7'b1101101; 
            4'h6: seg_out = 7'b1111101; 
            4'h7: seg_out = 7'b0000111; 
            4'h8: seg_out = 7'b1111111; 
            4'h9: seg_out = 7'b1101111; 
            4'hA: seg_out = 7'b1110111; 
            4'hB: seg_out = 7'b1111100; 
            4'hC: seg_out = 7'b0111001; 
            4'hD: seg_out = 7'b1011110; 
            4'hE: seg_out = 7'b1111001; 
            4'hF: seg_out = 7'b1110001; 
            default: seg_out = 7'b0000000; 
        endcase
    end
endmodule

module top_counter_visualized (
    input wire clk,
    input wire rst_n,
    input wire en,
    output wire [7:0] raw_count,    
    output wire [6:0] seg_display_low, 
    output wire [6:0] seg_display_high 
);

    wire tc_low; 

    counter_4bit lower_nibble (
        .clk(clk),
        .rst_n(rst_n),
        .en(en),
        .count(raw_count[3:0]),
        .tc(tc_low)
    );

    counter_4bit upper_nibble (
        .clk(clk),
        .rst_n(rst_n),
        .en(tc_low),
        .count(raw_count[7:4]),
        .tc() 
    );

    seven_seg_decoder display_low (
        .bin_in(raw_count[3:0]),
        .seg_out(seg_display_low)
    );

    seven_seg_decoder display_high (
        .bin_in(raw_count[7:4]),
        .seg_out(seg_display_high)
    );

endmodule
```
